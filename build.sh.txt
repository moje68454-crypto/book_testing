#!/usr/bin/env bash

# Install system dependencies
apt-get update
apt-get install -y python3 python3-pip build-essential

# Install Cython (required by some Node packages like tokenizers)
pip3 install cython

# Install Node dependencies
pnpm install  # or npm install if you're using npm

# Build your app (optional, if needed)
pnpm build  # or npm run build
